		var editor = ace.edit("editor");
		//editor.setValue($('#'+value).val()); 
		editor.setTheme("ace/theme/chrome");
		editor.getSession().setMode("ace/mode/html");
		editor.setAutoScrollEditorIntoView(true);
		editor.setShowPrintMargin(false);
		editor.setOption("enableEmmet", true); 
		editor.setOption('enableBasicAutocompletion',true);
		editor.setAutoScrollEditorIntoView(true);
				
		editor.getSession().on("change", function () {
			var code = editor.getSession().getValue().trim();
			localStorage.setItem('code',code);
			$('.preview').html(code);
		});

	   	$(document).ready(function() {
	   		editor.focus();
	   		//set value to ace editor
			if(editor.getSession().getValue() != null){
				var code = localStorage.getItem('code');
				$('.preview').html(code);
			}
			editor.setValue(localStorage.getItem('code'));
	   		
	   		//append the character cout and append html into output box 
			document.getElementById('editor').addEventListener('keyup',function(){
				show_stats();
				init_sliders();

			});

	   		
	   	});
	   	//get the line no and total line with character count
		function show_stats(){
			var currline = editor.getSelectionRange().start.row;
			var wholelinetxt = editor.getSession().getLine(currline);
			var current_row_characters = wholelinetxt.toString().trim().length;
			var data = editor.getSession().getValue().trim().length;
			var totalLines = editor.getSession().getLength();

			$('.editor-data').html('Line '+(currline+1)+' of '+totalLines+' ,Char '+current_row_characters);
		}
